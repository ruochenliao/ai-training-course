## 界面概述

Text2SQL系统前端界面基于React和Next.js构建，提供直观的自然语言查询输入、实时查询处理反馈、结果展示和数据可视化功能。界面设计注重用户体验，支持流式响应、多种数据展示方式和交互式操作。

## 核心界面组件

### 1. 智能查询输入界面
- 自然语言查询输入框
- 数据库连接配置面板
- 查询历史和模板选择
- 智能提示和自动补全

### 2. 实时处理状态界面
- 查询处理进度条
- 各阶段状态指示器
- 实时日志和反馈信息
- 取消操作控制

### 3. 结果展示界面
- 数据表格展示
- SQL语句显示和解释
- 数据统计信息
- 导出功能

### 4. 数据可视化界面
- 多种图表类型支持
- 交互式图表配置
- 可视化推荐
- 图表导出和分享

// WebSocket Hook
function useText2SQLWebSocket() {
  const [isConnected, setIsConnected] = useState(false)
  const [queryResults, setQueryResults] = useState<QueryResult[]>([])
  const [queryProgress, setQueryProgress] = useState<QueryProgress | null>(null)
  const [currentQueryId, setCurrentQueryId] = useState<string | null>(null)
  const [error, setError] = useState<string | null>(null)
  const wsRef = useRef<WebSocket | null>(null)
  const { toast } = useToast()

  const connect = useCallback(() => {
    try {
      const wsUrl = `ws://localhost:8000/api/v1/text2sql/websocket?session_id=${Date.now()}`
      wsRef.current = new WebSocket(wsUrl)

      wsRef.current.onopen = () => {
        setIsConnected(true)
        setError(null)
        toast({
          title: "连接成功",
          description: "WebSocket连接已建立"
        })
      }

      wsRef.current.onmessage = (event) => {
        try {
          const message = JSON.parse(event.data)
          handleWebSocketMessage(message)
        } catch (err) {
          console.error('解析WebSocket消息失败:', err)
        }
      }

      wsRef.current.onclose = () => {
        setIsConnected(false)
        toast({
          title: "连接断开",
          description: "WebSocket连接已断开",
          variant: "destructive"
        })
      }

      wsRef.current.onerror = (error) => {
        setError('WebSocket连接错误')
        toast({
          title: "连接错误",
          description: "WebSocket连接失败",
          variant: "destructive"
        })
      }
    } catch (err) {
      setError('无法建立WebSocket连接')
    }
  }, [])

  const disconnect = useCallback(() => {
    if (wsRef.current) {
      wsRef.current.close()
      wsRef.current = null
    }
  }, [])

  const sendQuery = useCallback((query: string, databaseConfig: DatabaseConfig) => {
    if (wsRef.current && wsRef.current.readyState === WebSocket.OPEN) {
      const message = {
        type: 'query_request',
        data: {
          query,
          database_config: databaseConfig
        }
      }
      wsRef.current.send(JSON.stringify(message))
    }
  }, [])

  const cancelQuery = useCallback(() => {
    if (wsRef.current && wsRef.current.readyState === WebSocket.OPEN && currentQueryId) {
      const message = {
        type: 'query_cancel',
        data: {
          query_id: currentQueryId
        }
      }
      wsRef.current.send(JSON.stringify(message))
    }
  }, [currentQueryId])

  const handleWebSocketMessage = (message: any) => {
    switch (message.type) {
      case 'connection_ack':
        console.log('连接确认:', message.data)
        break
      
      case 'query_start':
        setCurrentQueryId(message.data.query_id)
        setQueryProgress({ stage: 'started', progress: 0, message: '查询已开始...' })
        break
      
      case 'query_progress':
        setQueryProgress(message.data.progress)
        break
      
      case 'query_complete':
        setQueryResults(prev => [...prev, message.data])
        setQueryProgress(null)
        setCurrentQueryId(null)
        toast({
          title: "查询完成",
          description: "SQL查询已成功执行"
        })
        break
      
      case 'query_error':
        setError(message.data.error)
        setQueryProgress(null)
        setCurrentQueryId(null)
        toast({
          title: "查询错误",
          description: message.data.error,
          variant: "destructive"
        })
        break
      
      case 'heartbeat':
        // 心跳消息，保持连接活跃
        break
      
      default:
        console.log('未知消息类型:', message.type)
    }
  }

  useEffect(() => {
    connect()
    return () => disconnect()
  }, [])

  return {
    isConnected,
    queryResults,
    queryProgress,
    currentQueryId,
    error,
    sendQuery,
    cancelQuery,
    connect,
    disconnect
  }
}

// 数据库配置表单组件
function DatabaseConfigForm({ 
  config, 
  onChange, 
  showPassword, 
  onTogglePassword 
}: {
  config: DatabaseConfig
  onChange: (config: DatabaseConfig) => void
  showPassword: boolean
  onTogglePassword: () => void
}) {
  const updateConfig = (field: keyof DatabaseConfig, value: any) => {
    onChange({ ...config, [field]: value })
  }

  return (
    <div className="space-y-4">
      <div>
        <Label htmlFor="db-type">数据库类型</Label>
        <Select value={config.type} onValueChange={(value) => updateConfig('type', value)}>
          <SelectTrigger>
            <SelectValue />
          </SelectTrigger>
          <SelectContent>
            <SelectItem value="mysql">MySQL</SelectItem>
            <SelectItem value="postgresql">PostgreSQL</SelectItem>
            <SelectItem value="sqlite">SQLite</SelectItem>
            <SelectItem value="snowflake">Snowflake</SelectItem>
          </SelectContent>
        </Select>
      </div>
      
      {config.type !== 'sqlite' && (
        <>
          <div className="grid grid-cols-2 gap-4">
            <div>
              <Label htmlFor="host">主机</Label>
              <Input
                id="host"
                value={config.host || ''}
                onChange={(e) => updateConfig('host', e.target.value)}
                placeholder="localhost"
              />
            </div>
            <div>
              <Label htmlFor="port">端口</Label>
              <Input
                id="port"
                type="number"
                value={config.port || ''}
                onChange={(e) => updateConfig('port', parseInt(e.target.value))}
                placeholder="3306"
              />
            </div>
          </div>
          
          <div>
            <Label htmlFor="database">数据库名</Label>
            <Input
              id="database"
              value={config.database || ''}
              onChange={(e) => updateConfig('database', e.target.value)}
              placeholder="database_name"
            />
          </div>
          
          <div>
            <Label htmlFor="username">用户名</Label>
            <Input
              id="username"
              value={config.username || ''}
              onChange={(e) => updateConfig('username', e.target.value)}
              placeholder="username"
            />
          </div>
          
          <div>
            <Label htmlFor="password">密码</Label>
            <div className="relative">
              <Input
                id="password"
                type={showPassword ? 'text' : 'password'}
                value={config.password || ''}
                onChange={(e) => updateConfig('password', e.target.value)}
                placeholder="password"
              />
              <Button
                type="button"
                variant="ghost"
                size="sm"
                className="absolute right-0 top-0 h-full px-3"
                onClick={onTogglePassword}
              >
                {showPassword ? <EyeOff className="h-4 w-4" /> : <Eye className="h-4 w-4" />}
              </Button>
            </div>

// 数据表格显示组件
function DataTableDisplay({ result }: { result: QueryResult }) {
  const [currentPage, setCurrentPage] = useState(1)
  const [pageSize, setPageSize] = useState(10)
  const [sortColumn, setSortColumn] = useState<string | null>(null)
  const [sortDirection, setSortDirection] = useState<'asc' | 'desc'>('asc')
  const [filterText, setFilterText] = useState('')n
  
  const { data, columns } = result.results
  
  // 过滤和排序数据
  const filteredData = data.filter(row => 
    Object.values(row).some(value => 
      String(value).toLowerCase().includes(filterText.toLowerCase())
    )
  )
  
  const sortedData = sortColumn 
    ? [...filteredData].sort((a, b) => {
        const aVal = a[sortColumn]
        const bVal = b[sortColumn]
        const comparison = aVal < bVal ? -1 : aVal > bVal ? 1 : 0
        return sortDirection === 'asc' ? comparison : -comparison
      })
    : filteredData
  
  // 分页数据
  const totalPages = Math.ceil(sortedData.length / pageSize)
  const startIndex = (currentPage - 1) * pageSize
  const paginatedData = sortedData.slice(startIndex, startIndex + pageSize)
  
  const handleSort = (column: string) => {
    if (sortColumn === column) {
      setSortDirection(sortDirection === 'asc' ? 'desc' : 'asc')
    } else {
      setSortColumn(column)
      setSortDirection('asc')
    }
  }
  
  return (
    <div className="space-y-4">
      {/* 工具栏 */}
      <div className="flex items-center justify-between">
        <div className="flex items-center space-x-2">
          <div className="relative">
            <Search className="h-4 w-4 absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400" />
            <Input
              placeholder="搜索数据..."
              value={filterText}
              onChange={(e) => setFilterText(e.target.value)}
              className="pl-10 w-64"
            />
          </div>
          
          <Select value={pageSize.toString()} onValueChange={(value) => setPageSize(parseInt(value))}>
            <SelectTrigger className="w-32">
              <SelectValue />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="10">10行</SelectItem>
              <SelectItem value="25">25行</SelectItem>
              <SelectItem value="50">50行</SelectItem>
              <SelectItem value="100">100行</SelectItem>
            </SelectContent>
          </Select>
        </div>
        
        <div className="text-sm text-gray-600">
          显示 {startIndex + 1}-{Math.min(startIndex + pageSize, sortedData.length)} 条，
          共 {sortedData.length} 条记录
        </div>
      </div>
      
      {/* 数据表格 */}
      <div className="border rounded-lg overflow-hidden">
        <ScrollArea className="h-96">
          <table className="w-full">
            <thead className="bg-gray-50 sticky top-0">
              <tr>
                {columns.map((column) => (
                  <th 
                    key={column}
                    className="px-4 py-3 text-left text-sm font-medium text-gray-900 cursor-pointer hover:bg-gray-100"
                    onClick={() => handleSort(column)}
                  >
                    <div className="flex items-center space-x-1">
                      <span>{column}</span>
                      {sortColumn === column && (
                        sortDirection === 'asc' 
                          ? <SortAsc className="h-4 w-4" />
                          : <SortDesc className="h-4 w-4" />
                      )}
                    </div>
                  </th>
                ))}
              </tr>
            </thead>
            <tbody className="divide-y divide-gray-200">
              {paginatedData.map((row, index) => (
                <tr key={index} className="hover:bg-gray-50">
                  {columns.map((column) => (
                    <td key={column} className="px-4 py-3 text-sm text-gray-900">
                      {String(row[column] ?? '')}
                    </td>
                  ))}
                </tr>
              ))}
            </tbody>
          </table>
        </ScrollArea>
      </div>
      
      {/* 分页控件 */}
      {totalPages > 1 && (
        <div className="flex items-center justify-between">
          <Button
            variant="outline"
            onClick={() => setCurrentPage(Math.max(1, currentPage - 1))}
            disabled={currentPage === 1}
          >
            上一页
          </Button>
          
          <div className="flex items-center space-x-2">
            {Array.from({ length: Math.min(5, totalPages) }, (_, i) => {
              const page = i + 1
              return (
                <Button
                  key={page}
                  variant={currentPage === page ? "default" : "outline"}
                  size="sm"
                  onClick={() => setCurrentPage(page)}
                >
                  {page}
                </Button>
              )
            })}
          </div>
          
          <Button
            variant="outline"
            onClick={() => setCurrentPage(Math.min(totalPages, currentPage + 1))}
            disabled={currentPage === totalPages}
          >
            下一页
          </Button>
        </div>
      )}
    </div>
  )
}

// 可视化显示组件
function VisualizationDisplay({ result }: { result: QueryResult }) {
  const [selectedChartType, setSelectedChartType] = useState(result.visualization.chart_type)
  const [chartConfig, setChartConfig] = useState(result.visualization.config)
  
  const { data, columns } = result.results
  
  // 准备图表数据
  const prepareChartData = (type: string) => {
    switch (type) {
      case 'bar':
      case 'line':
        return {
          labels: data.map((_, index) => `Row ${index + 1}`),
          datasets: columns.map((column, index) => ({
            label: column,
            data: data.map(row => row[column]),
            backgroundColor: `hsl(${index * 60}, 70%, 50%)`,
            borderColor: `hsl(${index * 60}, 70%, 40%)`,
            borderWidth: 1
          }))
        }
      
      case 'pie':
      case 'doughnut':
        const firstColumn = columns[0]
        const secondColumn = columns[1] || columns[0]
        return {
          labels: data.map(row => row[firstColumn]),
          datasets: [{
            data: data.map(row => row[secondColumn]),
            backgroundColor: data.map((_, index) => `hsl(${index * 40}, 70%, 50%)`)
          }]
        }
      
      default:
        return { labels: [], datasets: [] }
    }
  }
  
  const chartData = prepareChartData(selectedChartType)
  
  const chartOptions: ChartOptions<any> = {
    responsive: true,
    maintainAspectRatio: false,
    plugins: {
      legend: {
        position: 'top' as const,
      },
      title: {
        display: true,
        text: '查询结果可视化'
      }
    },
    scales: selectedChartType === 'pie' || selectedChartType === 'doughnut' ? {} : {
      y: {
        beginAtZero: true
      }
    }
  }
  
  const renderChart = () => {
    switch (selectedChartType) {
      case 'bar':
        return <Bar data={chartData} options={chartOptions} />
      case 'line':
        return <Line data={chartData} options={chartOptions} />
      case 'pie':
        return <Pie data={chartData} options={chartOptions} />
      case 'doughnut':
        return <Doughnut data={chartData} options={chartOptions} />
      default:
        return <div className="text-center text-gray-500 py-8">不支持的图表类型</div>
    }
  }
  
  return (
    <div className="space-y-4">
      {/* 图表类型选择 */}
      <div className="flex items-center space-x-4">
        <Label>图表类型:</Label>
        <Select value={selectedChartType} onValueChange={setSelectedChartType}>
          <SelectTrigger className="w-48">
            <SelectValue />
          </SelectTrigger>
          <SelectContent>
            <SelectItem value="bar">柱状图</SelectItem>
            <SelectItem value="line">折线图</SelectItem>
            <SelectItem value="pie">饼图</SelectItem>
            <SelectItem value="doughnut">环形图</SelectItem>
          </SelectContent>
        </Select>
        
        <Button variant="outline" size="sm">
          <Download className="h-4 w-4 mr-2" />
          导出图表
        </Button>
      </div>
      
      {/* 推荐的可视化方案 */}
      {result.visualization.recommendations && result.visualization.recommendations.length > 0 && (
        <Card>
          <CardHeader>
            <CardTitle className="text-sm">推荐的可视化方案</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="flex flex-wrap gap-2">
              {result.visualization.recommendations.map((rec: any, index: number) => (
                <Badge 
                  key={index}
                  variant="outline"
                  className="cursor-pointer hover:bg-blue-50"
                  onClick={() => setSelectedChartType(rec.type)}
                >
                  {rec.name}
                </Badge>
              ))}
            </div>
          </CardContent>
        </Card>
      )}
      
      {/* 图表显示区域 */}
      <Card>
        <CardContent className="p-6">
          <div className="h-96">
            {renderChart()}
          </div>
        </CardContent>
      </Card>
    </div>
  )

## 部署配置

### Docker配置

```dockerfile
# Dockerfile
FROM node:18-alpine AS base

# Install dependencies only when needed
FROM base AS deps
RUN apk add --no-cache libc6-compat
WORKDIR /app

# Install dependencies based on the preferred package manager
COPY package.json yarn.lock* package-lock.json* pnpm-lock.yaml* ./
RUN \
  if [ -f yarn.lock ]; then yarn --frozen-lockfile; \
  elif [ -f package-lock.json ]; then npm ci; \
  elif [ -f pnpm-lock.yaml ]; then yarn global add pnpm && pnpm i --frozen-lockfile; \
  else echo "Lockfile not found." && exit 1; \
  fi

# Rebuild the source code only when needed
FROM base AS builder
WORKDIR /app
COPY --from=deps /app/node_modules ./node_modules
COPY . .

# Next.js collects completely anonymous telemetry data about general usage.
ENV NEXT_TELEMETRY_DISABLED 1

RUN yarn build

# Production image, copy all the files and run next
FROM base AS runner
WORKDIR /app

ENV NODE_ENV production
ENV NEXT_TELEMETRY_DISABLED 1

RUN addgroup --system --gid 1001 nodejs
RUN adduser --system --uid 1001 nextjs

COPY --from=builder /app/public ./public

# Set the correct permission for prerender cache
RUN mkdir .next
RUN chown nextjs:nodejs .next

# Automatically leverage output traces to reduce image size
COPY --from=builder --chown=nextjs:nodejs /app/.next/standalone ./
COPY --from=builder --chown=nextjs:nodejs /app/.next/static ./.next/static

USER nextjs

EXPOSE 3000

ENV PORT 3000
ENV HOSTNAME "0.0.0.0"

CMD ["node", "server.js"]
```
